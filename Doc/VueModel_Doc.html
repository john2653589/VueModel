#Easy to use - VueModel

Basic Setting： 1.new 2.VueInit 3.AddVue 4.AddSubmit
Call Api： 1.Ajax 2.Submit
AddV-DOM： 1.AddV_Input 2.AddV_Text 3.AddV_Button 4.AddV_Select

#Basic Setting

# 1.New 建構子
方法 VueModel()
說明 建構子
參數 _Url, _PageData, _VueOptions, _ElementName, IsMountedShow
參數解釋 預設Url, 頁面參數, 預設Vue設定, 綁定ElementId, 初始化執行Show()

# 使用方法

 
var Update = new VueModel('api/GetData', {}, {}, '__PageContent:__', true);

# 2.VueInit 初始化
方法 VueInit()
說明 初始化VueModel
參數 無
參數解釋 無

# 使用方法

 
var Update = new VueModel('api/GetData', {}, {}, '__PageContent:__', true).VueInit();

# 3.AddVue 加入[HttpGet]Key
方法 AddVue()
說明 加入一個Key至Vue存放區，並加入[HttpGet]Url至HttpGet存放區
參數 Key, Url
參數解釋 Key名稱, [HttpGet]Url

# 使用方法

 
var Update = new VueModel('api/GetData', {}, {}, '__PageContent:__', true)
        .AddVue('GetData','api/GetData');

# 4.AddSubmit 加入[HttpPost]Key
方法 AddSubmit()
說明 加入[HttpPost]Url至HttpPost存放區
參數 Key, Url, OnSuccess, OnError, OnComplate
參數解釋 Key名稱, [HttpGet]Url, Ajax 成功, Ajax 錯誤, Ajax 完成,

# 使用方法

 
var Update = new VueModel('api/GetData', {}, {}, '__PageContent:__', true)
        .AddSubmit('SendData','api/SendData',() => SuccessFunc());

#Call Api

# 1.Ajax HttpGet
方法 Ajax()
說明 發送 HttpGet 請求並執行VueInit()初始化
參數 SendData, Key, MethodType, OnSuccess, OnError, OnComplate
參數解釋 傳送參數, Url Key, Api類型, Ajax 成功, Ajax 錯誤, Ajax 完成,

# 使用方法

 
var Update = new VueModel('api/GetData', {}, {}, '__PageContent:__', true)
        .Ajax({ Id: 'abc-def-ghi-jkl' }, 'GetData', 'GET', () => SuccessFunc());

# 使用方法2

 
var Update = new VueModel('api/GetData', {}, {}, '__PageContent:__', true)
        .Ajax();
                    //SenData 預設 undefined
                    //Key 預設 Url 'api/GetData'
                    //MethodType 預設 'GET'

# 2.Submit HttpPost
方法 Submit()
說明 發送 HttpPost 請求並執行SuccessBackPage()，如不為undefined
參數 Key, MethodType, SendParam, _OnSuccess, _OnError, _OnComplate
參數解釋 Url Key, Api類型, 傳送參數, Ajax 成功, Ajax 錯誤, Ajax 完成,

# 使用方法

 
var Update = new VueModel('', {}, {}, '__PageContent:__', true)
        .Submit('Login', 'POST', Update.Result, () => SuccessFunc());

# 使用方法2

 
var Update = new VueModel('', {}, {}, '__PageContent:__', true)
        .Submit();
                    //Key 預設 Url，即 AddSubmit(undefined, 'api/Login')
                    //MethodType 預設 'POST'
                    //SendParam 預設 VueModel.Result

#AddV-DOM

# 1.Input
方法 AddV_Input()
說明 將InputId綁定至Vue存放區，使用 v-model 屬性
參數 InputId, InputV_Model
參數解釋 DOM 物件 id, 綁定資料來源，預設為Result.{InputId},

# 使用方法

 
var Update = new VueModel('api/GetData', {}, {}, '__PageContent:__', true)
        .AddV_Input('__UserName__', 'Result.UserName');

# 使用方法2

 
var Update = new VueModel('api/GetData', {}, {}, '__PageContent:__', true)
        .AddV_Input('__UserName__');
                    //InputV_Model 預設為 Result.{InputId} 即 Result.UserName

# 2.Text
方法 AddV_Text()
說明 將ObjectId綁定至Vue存放區，使用 v-text 屬性
參數 ObjectId, Key
參數解釋 DOM 物件 id, 綁定資料來源，預設為Result.{ObjectId},

# 使用方法

 
var Update = new VueModel('api/GetData', {}, {}, '__PageContent:__', true)
        .AddV_Text('__UserName__', 'Result.UserName');

# 使用方法2

 
var Update = new VueModel('api/GetData', {}, {}, '__PageContent:__', true)
        .AddV_Text('__UserName__');
                    //Key 預設為 Result.{ObjectId} 即 Result.UserName(自動移除'_')

# 3.Button
方法 AddV_Button()
說明 將ButtonId綁定至Vue存放區，並使用 AddFunction 方法加入 click 事件
參數 ButtonId, ClickResult, FuncKey
參數解釋 DOM 物件 id, 綁定 click 事件, click 事件名稱 預設為 Func{ButtonId}_Click,

# 使用方法

 
var Update = new VueModel('api/GetData', {}, {}, '__PageContent:__', true)
        .AddV_Button('__BtnSubmit__', Update.Submit());

# 4.Select
方法 AddV_Select()
說明 將SelectId綁定至Vue存放區，使用 AddVue() 方法加入 Url 網址，使用 AddV_SelectBind() 綁定 Select 欄位
參數 SelectId, Url, V_Value, V_Text, V_Model,
參數解釋 DOM 物件 id, 設定網址, Option值, Option顯示字, 選取Option存放區,

# 使用方法

 
var Update = new VueModel('api/GetData', {}, {}, '__PageContent:__', true)
        .AddV_Select('__SelectCity__', 'api/GetCity', 'CityId', 'CityName', 'Result.CityId');

# 使用方法2

 
var Update = new VueModel('api/GetData', {}, {}, '__PageContent:__', true)
        .AddV_Select('__SelectCity__', 'api/GetCity', 'CityId', 'CityName');
                    //V_Model 預設為 Result.{V_Value} 即 Result.CityId
